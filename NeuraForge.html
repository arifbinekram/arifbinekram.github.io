import React, { useState, useEffect } from 'react';
import { fetchData } from './api';

function App() {
  const [data, setData] = useState([]);
  
  useEffect(() => {
    fetchData().then(result => {
      setData(result);
    });
  }, []);
  
  return (
    <div className="app">
      {data.map(item => (
        <div key={item.id}>{item.name}</div>
      ))}
    </div>
  );
}
